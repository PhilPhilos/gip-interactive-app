    const observer = new IntersectionObserver((entries) => {

        entries.forEach(entry => {

            if (entry.isIntersecting) {

                navLinks.forEach(link => {

                    link.classList.remove('active');

                    if (link.getAttribute('href').substring(1) === entry.target.id) {

                        link.classList.add('active');

                    }

                });

            }

        });

    }, { threshold: 0.5 });


    sections.forEach(section => {

        observer.observe(section);

    });

    function runSimulation(step) {

        const selector = document.getElementById('claim-selector');

        const claimIndex = parseInt(selector.value);

        const selectedClaim = claims[claimIndex];

        const output = document.getElementById('analysis-output');

        const funnelSteps = [

            document.getElementById('funnel-1'),

            document.getElementById('funnel-2'),

            document.getElementById('funnel-3')

        ];


        funnelSteps.forEach(fs => {

            fs.style.borderColor = 'transparent';

            fs.style.backgroundColor = 'white';

        });


        if (step === 0) { // Full analysis

            output.innerHTML = selectedClaim.analysis.join('<br><br>');

            updateChart(selectedClaim.scores);

            funnelSteps[0].style.borderColor = '#818cf8'; // indigo-400

            funnelSteps[0].style.backgroundColor = '#eef2ff'; // indigo-50

            funnelSteps[1].style.borderColor = '#5eead4'; // teal-300

            funnelSteps[1].style.backgroundColor = '#f0fdfa'; // teal-50

            funnelSteps[2].style.borderColor = '#fcd34d'; // amber-300

            funnelSteps[2].style.backgroundColor = '#fffbeb'; // amber-50

        } else { // Step-by-step

            output.innerHTML = selectedClaim.analysis[step - 1];

            updateChart(getScoresForStep(selectedClaim.scores, step));

             if (step >= 1) {

                funnelSteps[0].style.borderColor = '#818cf8';

                funnelSteps[0].style.backgroundColor = '#eef2ff';

            }

            if (step >= 2) {

                funnelSteps[1].style.borderColor = '#5eead4';

                funnelSteps[1].style.backgroundColor = '#f0fdfa';

            }
